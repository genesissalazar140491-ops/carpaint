📊 Configuración Google Sheets
Para activar el respaldo en tiempo real, necesitas:
1. Crear un Google Sheet vacío
2. En el Sheet, ir a Extensiones → Apps Script
3. Pegar el script que aparece abajo y guardarlo como Web App
4. Copiar la URL del Web App aquí abajo
// ⚠️ REEMPLAZA TODO el código del Apps Script con esto y vuelve a publicar como Web App:
function doPost(e) {
  try {
  const data = JSON.parse(e.postData.contents);
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const cls = (name) => { let s=ss.getSheetByName(name); if(s) s.clearContents(); return s||ss.insertSheet(name); };
  const get = (name) => ss.getSheetByName(name)||ss.insertSheet(name);

  if(data.type === 'hist') {
    let sh=get('Historial');
    if(sh.getLastRow()===0) sh.appendRow(['#','Fecha','Hora','Cliente','Tipo','Producto','Fracc','Cant','Precio$','Total$','TotalBs','Metodo','Colorista','Obs']);
    data.rows.forEach(r => sh.appendRow(r));

  } else if(data.type === 'full') {
    // ── HISTORIAL ──
    let sh=cls('Historial');
    sh.appendRow(['#','Fecha','Hora','Cliente','Tipo','Producto','Fracc','Cant','Precio$','Total$','TotalBs','Metodo','Colorista','Obs']);
    data.hist.forEach((r,i)=>sh.appendRow([i+1,r.fecha,r.hora||'',r.cliente,r.tipo,r.prod,r.frac,r.cant,r.price,r.total,r.totalBs||0,r.metodo,r.colorista||'',r.obs||'']));

    // ── CAJA ──
    let sc=cls('Caja');
    sc.appendRow(['Método','Monto USD','Monto Bs']);
    const ts=data.tasa||1;
    const c=data.caja;
    sc.appendRow(['📱 Pago Móvil', c.movil||0, (c.movil||0)*ts]);
    sc.appendRow(['🇻🇪 Bs Efectivo', c.bs_ef||0, (c.bs_ef||0)*ts]);
    sc.appendRow(['💵 USD Efectivo', c.usd||0, '']);
    sc.appendRow(['🏦 Zelle', c.zelle||0, '']);
    sc.appendRow(['₿ Binance', c.binance||0, '']);
    sc.appendRow(['📋 Crédito', c.credito||0, '']);
    sc.appendRow(['💳 Abonos', c.abonos||0, '']);
    const tot=(c.movil||0)+(c.bs_ef||0)+(c.usd||0)+(c.zelle||0)+(c.binance||0)+(c.credito||0)+(c.abonos||0);
    sc.appendRow(['','','']);
    sc.appendRow(['TOTAL DÍA', tot, '']);

    // ── TALLER ──
    let st=cls('Taller');
    st.appendRow(['Fecha','Cliente','Producto','Fracción','Cant','Total $']);
    data.taller.forEach(r=>st.appendRow([r.fecha,r.cliente,r.prod,r.frac,r.cant,r.total]));
    st.appendRow(['','','','','TOTAL',data.totalTaller]);

    // ── DANIEL COLORISTA ──
    let sd=cls('Daniel_Colorista');
    sd.appendRow(['Fecha','Cliente','Color / Producto','Fracción','Monto $']);
    data.colorista.forEach(r=>sd.appendRow([r.fecha,r.cliente,r.color,r.frac,r.monto]));
    sd.appendRow(['','','','TOTAL VENTAS',data.totalColorista]);
    sd.appendRow(['','','','COMISIÓN 10%',data.comisionDaniel]);

    // ── INVENTARIO CRÍTICO ──
    let si=cls('Inventario_Critico');
    si.appendRow(['Cód.','Producto','Marca','Stock Actual','Stock Mín','Estado','¿Pedir?']);
    data.inv_critico.forEach(r=>si.appendRow([r.cod,r.name,r.marca,r.stock,r.stockMin,r.estado,'SÍ']));
    si.appendRow(['','','Total a reponer:',data.inv_critico.length,'','','']);
  }
  return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})).setMimeType(ContentService.MimeType.JSON);
  }
}
function doGet(e) { return ContentService.createTextOutput('CarPaint POS OK'); }
